 24 require.def('antie/widgets/carousel/orientations/vertical',
 25     [
 26         'antie/class',
 27         'antie/events/keyevent'
 28     ],
 29     /**
 30      * Class to encapsulate any data specific to a vertical orientation
 31      * @name antie.widgets.carousel.orientations.Vertical
 32      * @extends antie.class
 33      * @class
 34      */
 35     function (Class, KeyEvent) {
 36         "use strict";
 37         var Vertical = Class.extend({
 38             dimension: function () {
 39                 return 'height';
 40             },
 41             edge: function () {
 42                 return 'top';
 43             },
 44             styleClass: function () {
 45                 return 'vertical';
 46             },
 47             defaultKeys: function () {
 48                 return {
 49                     PREVIOUS: KeyEvent.VK_UP,
 50                     NEXT: KeyEvent.VK_DOWN
 51                 };
 52             }
 53         });
 54 
 55         return new Vertical();
 56     }
 57 );
