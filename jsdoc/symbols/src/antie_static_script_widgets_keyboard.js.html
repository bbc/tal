 27 require.def('antie/widgets/keyboard',
 28 	[
 29 		'antie/widgets/grid',
 30 		'antie/widgets/button',
 31 		'antie/widgets/label',
 32 		'antie/events/keyevent',
 33 		'antie/events/textchangeevent'
 34 	],
 35 	function(Grid, Button, Label, KeyEvent, TextChangeEvent) {
 36 		/**
 37 		 * On-screen keyboard widget.
 38 		 * @name antie.widgets.Keyboard
 39 		 * @class
 40 		 * @extends antie.widgets.Grid
 41 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 42 		 * @param {Integer} cols The number of columns of keys on the keyboard.
 43 		 * @param {Integer} rows The number of rows of keys on the keyboard.
 44 		 * @param {String} keys A string of characters which make up the keys, starting top-left and working row-by-row to the bottom-right.
 45 	 	 *			Special characters include:
 46 		 *				"-"	DEL
 47 		 *				" "	SPACE
 48 		 *				"_"	<spacer>
 49 		 */
 50 		var Keyboard = Grid.extend(/** @lends antie.widgets.Keyboard.prototype */ {
 51 			/**
 52 			 * @constructor
 53 			 * @ignore
 54 			 */
 55 			init: function(id, cols, rows, keys) {
 56 				this._super(id, cols, rows);
 57 
 58 				this.addClass('keyboard');
 59 
 60 				var self = this;
 61 				var device = self.getCurrentApplication().getDevice();
 62 				var multitap = device.getConfig().input.multitap;
 63 
 64 				this._focussedCharacter = null;
 65 				this._currentText = "";
 66 				this._letterButtons = [];
 67 				this._keys = keys;
 68 				this._multiTap = false;
 69 				this._multiTapTimeout = null;
 70 				this._multiTapLastKey = null;
 71 				this._multiTapLastKeyIndex = 0;
 72 				this._capitalisation = Keyboard.CAPITALISATION_UPPER;
 73 				this._maximumLength = null;
 74 
 75 				// Private function to append a properly-capitalised character to the end of the string.
 76 				function appendCharacter(letter) {
 77 					// allow no more characters to be appended if a maximum length has been reached
 78 					if((self._maximumLength !== null) && (self._currentText.length >= self._maximumLength)) {
 79 						return false;
 80 					}
 81 
 82 					if(	(self._capitalisation != Keyboard.CAPITALISATION_LOWER)
 83 						&& (
 84 							(self._capitalisation == Keyboard.CAPITALISATION_UPPER)
 85 							|| self._currentText.length == 0
 86 							|| self._currentText[self._currentText.length-1] == ' '
 87 						)
 88 					) {
 89 						letter = letter.toUpperCase();
 90 					} else {
 91 						letter = letter.toLowerCase();
 92 					}
 93 					self._currentText += letter;
 94 
 95 					correctTitleCase();
 96 	
 97 					return true;
 98 				}
 99 				function correctTitleCase() {
100 					if(self._capitalisation == Keyboard.CAPITALISATION_TITLE) {
101 						self._currentText = self._currentText.replace(Keyboard.LAST_WORD_REGEXP, function(match) {
102 							match = match.substring(0, 1).toUpperCase() + match.substring(1);
103 							return match.replace(Keyboard.SHORT_WORD_REGEXP, function(match) { return match.toLowerCase(); });
104 						});
105 						self._currentText = self._currentText.substring(0, 1).toUpperCase() + self._currentText.substring(1);
106 					}
107 				}
108 
109 				// Select event handler for buttons within the keyboard
110 				this.addEventListener('select', function(evt) {
111 					var letter = evt.target.getDataItem();
112 					var changed = false;
113 					switch(letter) {
114 						case "DEL":
115 							if(self._currentText.length > 0) {
116 								self._currentText = self._currentText.substring(0, self._currentText.length - 1);
117 								correctTitleCase();
118 								changed = true;
119 							}
120 							break;
121 						case "SPACE":
122 							changed = appendCharacter(" ");
123 							break;
124 						default:
125 							changed = appendCharacter(letter);
126 							break;
127 					}
128 
129 					if(changed) {
130 						self._updateClasses();
131 						self.bubbleEvent(new TextChangeEvent(self, self._currentText, evt.target, false));
132 					}
133 				});
134 
135 				// Event listener to handle keyboard/numeric button press events.
136 				this.addEventListener('keydown', function(evt) {
137 					if(evt.keyChar) {
138 						evt.stopPropagation();
139 						// If the device supports multitap, multitap is enabled and a number is pressed...
140 						if(multitap && self._multiTap && /[0-9]/.test(evt.keyChar)) {
141 							if(self._multiTapTimeout) {
142 								clearTimeout(self._multiTapTimeout);
143 							}
144 
145 							var chars = multitap[evt.keyChar];
146 							if((evt.keyChar == self._multiTapLastKey) && self._multiTapTimeout) {
147 								self._currentText = self._currentText.substring(0, self._currentText.length - 1);
148 							} else {
149 								self._multiTapLastKeyIndex = -1;
150 							}
151 
152 							// Find the next character for the pressed key that's available on this keyboard
153 							do {
154 								self._multiTapLastKeyIndex++;
155 								if(self._multiTapLastKeyIndex >= chars.length) self._multiTapLastKeyIndex = 0;
156 							} while(!self._letterButtons[chars[self._multiTapLastKeyIndex]]);
157 
158 							self._focussedCharacter = chars[self._multiTapLastKeyIndex];
159 							self._letterButtons[chars[self._multiTapLastKeyIndex]].focus();
160 							appendCharacter(chars[self._multiTapLastKeyIndex]);
161 
162 							self._multiTapLastKey = evt.keyChar;
163 
164 							self._updateClasses();
165 
166 							// Fire a text change event, but notify listeners that it may change due to being multitap
167 							self.bubbleEvent(new TextChangeEvent(self, self._currentText, null, true));
168 
169 							self._multiTapTimeout = setTimeout(function() {
170 								self._multiTapTimeout = null;
171 								// Fire a new text change event to notify listeners that the multi-tap timeout has finished
172 								self.bubbleEvent(new TextChangeEvent(self, self._currentText, null, false));
173 							}, 1000);
174 						} else {
175 							// Select and focus the button on the keyboard for the pressed key
176 							var button = self._letterButtons[evt.keyChar];
177 							if(button) {
178 								self._focussedCharacter = evt.keyChar;
179 								button.focus();
180 								button.select();
181 							}
182 						}
183 					} else if(evt.keyCode == KeyEvent.VK_BACK_SPACE) {
184 						if(self._currentText.length > 0) {
185 							self._currentText = self._currentText.substring(0, self._currentText.length - 1);
186 							correctTitleCase();
187 
188 							self._updateClasses();
189 
190 							self.bubbleEvent(new TextChangeEvent(self, self._currentText, null, false));
191 						}
192 					} else if(evt.keyCode == KeyEvent.VK_RIGHT) {
193 						if(self._multiTapTimeout) {
194 							self._multiTapTimeout = null;
195 							// Fire a new text change event to notify listeners that the multi-tap timeout has finished
196 							self.bubbleEvent(new TextChangeEvent(self, self._currentText, null, false));
197 							evt.stopPropagation();
198 						}
199 					}
200 				});
201 
202 				// Populate the grid (ourself) with the key Buttons
203 				for(var col = 0; col < 14; col++) {
204 					for(var row = 0; row < 3; row++) {
205 						var letter = this._keys[(row*14)+col];
206 						if(letter == " ") { letter = "SPACE" }
207 						else if(letter == "-") { letter = "DEL" }
208 						else if(letter == "_") { continue };
209 
210 						var button = new Button(this.id + '_' + letter + "_" + col + "_" + row);
211 						button.setDataItem(letter);
212 						button.addClass('key'+letter);
213 						button.appendChildWidget(new Label(letter));
214 
215 						this._letterButtons[this._keys[(row*14)+col]] = button;
216 
217 						this.setWidgetAt(col, row, button);
218 					}
219 				}
220 			},
221 			/**
222 			 * Sets whether to support multi-tap on this keyboard. Note: the device must also support it.
223 			 * @param {Boolean} multiTap Pass <code>true</code> to enable multi-tap.
224 			 */
225 			setMultiTap: function(multiTap) {
226 				this._multiTap = multiTap;
227 			},
228 			/**
229 			 * Gets whether to multi-tap is supported by this keyboard.
230 			 * @returns Boolean <code>true</code> if multi-tap is supported by this keyboard.
231 			 */
232 			getMultiTap: function() {
233 				return this._multiTap;
234 			},
235 			/**
236 			 * Sets the current text entered/to-be-edited by this keyboard.
237 			 * @param {String} text String to be edited by this keyboard.
238 			 */
239 			setText: function(text) {
240 				this._currentText = text;
241 				this._updateClasses();
242 			},
243 			/**
244 			 * Gets the text entered/edited by this keyboard.
245 			 * @returns The text entered/edited by this keyboard.
246 			 */
247 			getText: function() {
248 				return this._currentText;
249 			},
250 			/**
251 			 * Sets the capitalisation mode of the keyboard. {@see Keyboard.CAPITALISATION_UPPER},
252 			 * {@see Keyboard.CAPITALISATION_LOWER} and {@see Keyboard.CAPITALISATION_TITLE}
253 			 * @param {Integer} capitalisation The capitalisation mode to use.
254 			 */
255 			setCapitalisation: function(capitalisation) {
256 				this._capitalisation = capitalisation;
257 			},
258 			/**
259 			 * Gets the current capitalisation mode of the keyboard.
260 			 * @returns The capitalisation mode of the keyboard.
261 			 */
262 			getCapitalisation: function() {
263 				return this._capitalisation;
264 			},
265 			/**
266 			 * Sets the active child widget as the button for the specified character.
267 			 * @param {String} character The character who's button should be the active child widget.
268 			 */
269 			setActiveChildKey: function(character) {
270 				this._focussedCharacter = character;
271 				this.setActiveChildWidget(this._letterButtons[character]);
272 			},
273 			focus: function() {
274 				this._letterButtons[this._focussedCharacter].focus();
275 			},
276 			setFocusToActiveChildKey: function(character){
277 				this.setActiveChildKey(character);
278 				this.focus();
279 			},
280 			/**
281 			 * Sets the maximum number of characters that can be entered
282 			 * @param {Integer} length The maxmimum number of characters that can be entered. Pass <code>null</code> to allow infinite characters.
283 			 */
284 			setMaximumLength: function(length) {
285 				this._maximumLength = length;
286 			},
287 			/**
288 			 * Gets the maximum number of characters that can be entered
289 			 * @returns The maximum number of characters that can be entered
290 			 */
291 			getMaximumLength: function() {
292 				return this._maximumLength;
293 			},
294 
295 			_updateClasses: function() {
296 				if((this._maximumLength !== null) && (this._currentText != null) && (this._currentText.length >= this._maximumLength)) {
297 					this.addClass('maxlength');
298 					// Move focus to the DEL key, as it's
299 					// the only one that can be used now.
300 					this.setActiveChildKey("-");
301 				} else {
302 					this.removeClass('maxlength');
303 				}
304 			}
305 		});
306 
307 		Keyboard.LAST_WORD_REGEXP = /([^\s]+)$/;
308 		Keyboard.SHORT_WORD_REGEXP = /^(to|is|in|on|it|and|at|by|a|an|of|if)$/i;
309 
310 		Keyboard.CAPITALISATION_UPPER = 0;
311 		Keyboard.CAPITALISATION_LOWER = 1;
312 		Keyboard.CAPITALISATION_TITLE = 2;
313 
314 		return Keyboard;
315 	}
316 );
