 27 require.def(
 28     'antie/devices/anim/css3/transitiondefinition',
 29     [
 30         'antie/class',
 31         'antie/devices/anim/css3/easinglookup'
 32     ],
 33     function(Class, EasingLookup) {
 34         "use strict";
 35         var TransitionDefinition;
 36         
 37         /*
 38          * Defines a class of transition in terms of its properties and their transition attributes.
 39          * It defines durations, delays and timing functions but not start and end points or callbacks 
 40          * (these are handled by TransitionEndPoints)
 41          */
 42         TransitionDefinition = Class.extend({
 43             init: function(){
 44                 this.properties = {};
 45             },
 46             
 47             setProperty: function(property, attributes){
 48                 if(attributes) {
 49                     this.properties[property] = {
 50                         duration: (attributes.duration !== undefined ? attributes.duration : TransitionDefinition.defaults.duration),
 51                         delay: (attributes.delay !== undefined ? attributes.delay : TransitionDefinition.defaults.delay),
 52                         timingFn: (attributes.timingFn !== undefined ? attributes.timingFn : TransitionDefinition.defaults.timingFn)
 53                     };
 54                 } else {
 55                     this.properties[property] = {
 56                         duration: (TransitionDefinition.defaults.duration),
 57                         delay: (TransitionDefinition.defaults.delay),
 58                         timingFn: (TransitionDefinition.defaults.timingFn)
 59                     };
 60                 }
 61             },
 62                         
 63             hasProperty: function(property) {
 64                 return this.properties.hasOwnProperty(property);
 65             },
 66             
 67             getProperties: function() {
 68                 var prop, propArr;
 69                 propArr = [];
 70                 for (prop in this.properties) {
 71                     if(this.hasProperty(prop)) {
 72                         propArr.push(prop);
 73                     }
 74                 }
 75                 return propArr;
 76             },
 77             
 78             removeProperty: function(property) {
 79                 if(this.hasProperty(property)) {
 80                     delete this.properties[property];
 81                     return true;
 82                 }
 83                 return false;
 84             },
 85             
 86             getPropertyDuration: function(property) {
 87                 return this._getPropertyAttribute(property, 'duration');
 88                 
 89             },
 90             
 91             getPropertyDelay: function(property) {
 92                 return this._getPropertyAttribute(property, 'delay');
 93             },
 94             
 95             getPropertyTimingFn: function(property) {
 96                 return this._getPropertyAttribute(property, 'timingFn');
 97             },
 98             
 99             setPropertyDuration: function(property, duration) {
100                 if (this.hasProperty(property)) {
101                     this.properties[property].duration = duration;
102                 } else {
103                     this.setProperty(property, {
104                         duration: duration
105                     });
106                 }
107             },
108             
109             setPropertyDelay: function(property, delay) {
110                 if (this.hasProperty(property)) {
111                     this.properties[property].delay = delay;
112                 } else {
113                     this.setProperty(property, {
114                         delay: delay
115                     });
116                 }
117             },
118             
119             setPropertyTimingFn: function(property, timingFn) {
120                 if (this.hasProperty(property)) {
121                     this.properties[property].timingFn = timingFn;
122                 } else {
123                     this.setProperty(property, {
124                         timingFn: timingFn
125                     });
126                 }
127             },
128             
129             addIn: function(transitionDefinition) {
130                 var property, addProps, i;
131                 addProps = transitionDefinition.getProperties();
132                 for (i = 0; i !== addProps.length; i += 1) {
133                     property = addProps[i];
134                     this.setProperty(property, {
135                         duration: transitionDefinition.getPropertyDuration(property),
136                         delay: transitionDefinition.getPropertyDelay(property),
137                         timingFn: transitionDefinition.getPropertyTimingFn(property)
138                     });
139                 }
140             },
141             
142             takeOut: function(transitionDefinition) {
143                 var subProps, i;
144                 subProps = transitionDefinition.getProperties();
145                 for (i = 0; i !== subProps.length; i += 1) {
146                     if(this.hasProperty(subProps[i])) {
147                         this.removeProperty(subProps[i]);
148                     }
149                 }
150             },
151             
152             areAllDurationsZero: function() {
153                 var property, subProps, i, allZeros;
154                 allZeros = true;
155                 subProps = this.getProperties();
156                 for (i = 0; i !== subProps.length; i += 1) {
157                     property = subProps[i];
158                     if(this.hasProperty(property)) {
159                         allZeros = allZeros && this.getPropertyDuration(property) === 0;
160                     }
161                 }
162                 return allZeros;
163             },
164             
165             _getPropertyAttribute: function(property, attribute) {
166                 if(this.hasProperty(property)) {
167                     return this.properties[property][attribute];
168                 }
169                 return undefined;
170             }
171         });
172            
173         TransitionDefinition.easingLookup = new EasingLookup();
174         
175         TransitionDefinition.defaults = {
176             duration: 840,
177             delay: 0,
178             timingFn: TransitionDefinition.easingLookup.easeInOut
179         };
180         
181         return TransitionDefinition;
182     }
183 );
