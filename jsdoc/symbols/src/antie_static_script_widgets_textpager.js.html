 27 require.def('antie/widgets/textpager',
 28 	[
 29 	 	'antie/widgets/label',
 30 	 	'antie/events/textpagechangeevent'
 31 	],
 32 	function(Label, TextPageChangeEvent) {
 33 		'use strict';
 34 
 35 		/**
 36 		 * The TextPager widget displays text. It computers its own size, allows scrolling and reports page numbers.
 37 		 * @name antie.widgets.TextPager
 38 		 * @class
 39 		 * @extends antie.widgets.Label
 40 		 * @requires antie.events.TextPageChangeEvent
 41 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 42 		 */
 43 		var TextPager = Label.extend(/** @lends antie.widgets.TextPager.prototype */{
 44 			/**
 45 			 * @constructor
 46 			 * @ignore
 47 			 */
 48 			init: function(id) {
 49 				this._super(id, "");
 50 				this.addClass('textpager');
 51 				this._page = 1;
 52 			},
 53 			/**
 54 			 * Renders the widget and any child widgets to device-specific output.
 55 			 * @param {antie.devices.Device} device The device to render to.
 56 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 57 			 */
 58 			render: function(device) {
 59 				var s = this._text;
 60 				
 61 				if(!this.outputElement) {
 62 					this.outputElement = device.createContainer(this.id, this.getClasses());
 63 					this.innerElement = device.createContainer(this.id + "_inner");
 64 					this.outputElement.appendChild(this.innerElement);
 65 				}
 66 				device.setElementContent(this.innerElement, s);
 67 				
 68 				return this.outputElement;
 69 			},
 70 			getPageCount: function () {
 71 				if (!this._paddingFudge) {
 72 					var el = this.outputElement;
 73 					this._paddingFudge = true;
 74 					this.textHeight = this.innerElement.clientHeight;
 75 					this.boxHeight = this.outputElement.clientHeight;
 76 					
 77 					this.lineHeight = this.getCurrentApplication().getLayout().textPager.lineHeight;
 78 					
 79 					this.lines_in_box = Math.floor(this.boxHeight / this.lineHeight);
 80 					this.lines_in_text = Math.floor(this.textHeight / this.lineHeight);
 81 					
 82 					this._pageCount = Math.ceil(this.lines_in_text / this.lines_in_box);
 83 					var extra = this.textHeight - (this.lines_in_text % this.lines_in_box);
 84 					this.innerElement.style.paddingBottom = this.lineHeight * extra + 'px';
 85 				}
 86 				return this._pageCount;
 87 			},
 88 			getCurrentPage: function () {
 89 				return this._page;
 90 			},
 91 			setPage: function (page) {
 92 				var el = this.outputElement;
 93 				this._page = page;
 94 				el.scrollTop = (page - 1) * this.lineHeight * this.lines_in_box;
 95 				this.bubbleEvent(new TextPageChangeEvent(this, page));
 96 			},
 97 			pageUp: function () {
 98 				var page = this.getCurrentPage();
 99 				if (page === 1) {
100 					return;
101 				} 
102 				this.setPage(page - 1);
103 			},
104 			pageDown: function () {
105 				var max = this.getPageCount();
106 				var page = this.getCurrentPage();
107 				if (page === max) {
108 					return;
109 				} 
110 				this.setPage(page + 1);
111 			},
112 
113 			setText: function(text) {
114 				this._super(text);
115 
116 				// Remove the bottom padding to allow the page count to be
117 				// recalculated.
118 				if(this.innerElement) {
119 					this.innerElement.style.paddingBottom = 0;
120 				}
121 				this._paddingFudge = false;
122 			}
123 		});
124 		return TextPager;
125 	}
126 );
