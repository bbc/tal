 27 require.def('antie/devices/googletv',
 28 	[
 29 		'antie/devices/browserdevice',
 30 		'antie/runtimecontext'
 31 	],
 32 	function(BrowserDevice, RuntimeContext) {
 33 		'use strict';
 34 
 35 		return BrowserDevice.extend({
 36 			init: function(config) {
 37 				// Change Application::getBestFitLayout to find the layout with a size
 38 				// closest to that of the browser resolution, then use CSS zoom to fit
 39 				// it to the exact screen size.
 40 				RuntimeContext.getCurrentApplication().getBestFitLayout = function() {
 41 					var _screenSize = this._device.getScreenSize();
 42 					var _layouts = this._device.getConfig().layouts;
 43 
 44 					// sort the layouts by closest to reported screen resolution first
 45 					_layouts.sort(function(a, b) {
 46 						var adiff = {
 47 								width: Math.abs(_screenSize.width - a.width),
 48 								height: Math.abs(_screenSize.height - a.height)
 49 						};
 50 						var bdiff = {
 51 								width: Math.abs(_screenSize.width - b.width),
 52 								height: Math.abs(_screenSize.height - b.height)
 53 						};
 54 						var	ad = (adiff.width*adiff.width) + (adiff.height*bdiff.height),
 55 							bd = (bdiff.width*bdiff.width) + (bdiff.height*bdiff.height);
 56 
 57 						if(ad == bd) return 0;
 58 						else if(ad < bd) return -1;
 59 						else return 1;
 60 					});
 61 
 62 					var _selected = _layouts[0];
 63 
 64 					// scale the closest layout to the exact screen resolution
 65 					document.body.style.zoom = (screen.width / _selected.width);
 66 
 67 					return _selected;
 68 				};
 69 
 70 				this._super(config);
 71 			}
 72 		});
 73 	}
 74 );
