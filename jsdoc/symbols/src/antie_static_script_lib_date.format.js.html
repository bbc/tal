  7 Date.prototype.format = function(format) {
  8 	var returnStr = '';
  9 	var replace = Date.replaceChars;
 10 	for (var i = 0; i < format.length; i++) {
 11 		var curChar = format.charAt(i);
 12 		if (i - 1 >= 0 && format.charAt(i - 1) == "\\") { 
 13 			returnStr += curChar;
 14 		}
 15 		else if (replace[curChar]) {
 16 			returnStr += replace[curChar].call(this);
 17 		} else if (curChar != "\\"){
 18 			returnStr += curChar;
 19 		}
 20 	}
 21 	return returnStr;
 22 };
