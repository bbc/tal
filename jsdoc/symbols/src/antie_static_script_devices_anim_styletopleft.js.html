 37 		function movesScroll( startLeft, startTop, changeLeft, changeTop, options ){
 38 		    var to, from;
 39 			if ((changeLeft === 0) && (changeTop === 0)) {
 40 					if (options.onComplete) {
 41 						options.onComplete();
 42 					}
 43 					return null;
 44 				}
 45 
 46 			if (this.getConfig().animationDisabled || options.skipAnim) {
 47 				if (options.to.left !== undefined) {
 48 					options.el.style.left = options.to.left + "px";
 49 				}
 50 				if (options.to.top !== undefined) {
 51 					options.el.style.top = options.to.top + "px";
 52 				}
 53 
 54 				if (options.onComplete) {
 55 					options.onComplete();
 56 				}
 57                 return null;
 58 			} else {
 59 				from = {};
 60 				if (startTop !== undefined) {
 61 					from.top = startTop + "px";
 62 				}
 63 				if (startLeft !== undefined) {
 64 					from.left = startLeft + "px";
 65 				}
 66 
 67 				to = {};
 68 				if (options.to.top !== undefined) {
 69 					to.top = (options.to.top) + "px";
 70 				}
 71 				if (options.to.left !== undefined) {
 72 					to.left = (options.to.left) + "px";
 73 				}
 74 
 75 				return this._tween({
 76 					el: options.el,
 77 					style: options.el.style,
 78 					from: from,
 79 					to: to,
 80 					offset: options.offset || 0,
 81 					easing: options.easing,
 82 					fps: options.fps,
 83 					duration: options.duration,
 84 					onComplete: options.onComplete
 85 				});
 86 			}
 87 		}

 89 		Device.prototype.scrollElementTo = function(options) {
 90 		    var startLeft, changeLeft, startTop, changeTop, newOptions;
 91 			// Helper function to return an object that inherits from the original
 92 			function inherit(original) {
 93 				function Inherited() {}
 94 				
 95 				Inherited.prototype = original;
 96 				return new Inherited();
 97 			}
 98 			
 99 			// Make a new modifiable options object inheriting from the original. Need to do this rather than
100 			// simply reverting any changes before returning, as the tweening library calls onComplete() which
101 			// may require an unchanged object.
102 			newOptions = inherit(options);
103 			if (options.to) {
104 				newOptions.to = inherit(options.to);
105 			}
106 
107 			// Check validity of call and use child element as the real target
108 			if (new RegExp("_mask$").test(options.el.id)) {
109 				if (options.el.childNodes.length === 0) {
110 					return null;
111 				}
112 				options.el.style.position = 'relative';
113 				newOptions.el = options.el.childNodes[0];
114 				newOptions.el.style.position = 'relative';
115 			} else {
116 				return null;
117 			}
118 			
119 			// Make a copy of the 'to' property, with the sign of the 'top' and 'left' properties flipped.
120 			if (options.to && options.to.left) {
121 				newOptions.to.left = -options.to.left;
122 			}
123 			if (options.to && options.to.top) {
124 				newOptions.to.top = -options.to.top;
125 			}
126 
127 			startLeft = newOptions.el.style.left.replace(/px/, '') || 0;
128 			changeLeft = (options.to.left !== undefined) ? (options.to.left - Math.abs(startLeft)) : 0;
129 			startTop = newOptions.el.style.top.replace(/px/, '') || 0;
130 			changeTop = (options.to.top !== undefined) ? (options.to.top - Math.abs(startTop)) : 0;
131 
132 			return movesScroll.apply( this, [ startLeft, startTop, changeLeft, changeTop, newOptions ] );
133 		};

135         /* documented in antie.devices.Device */
136 		Device.prototype.moveElementTo = function(options) {
137 		    var startLeft, changeLeft, startTop, changeTop;
138 			// Performance consideration: if left or top is null they are ignored to prevent the additional
139 			// work animating them.
140 
141 			startLeft = parseInt(options.el.style.left.replace(/px|em|pt/,""), 10) || 0;
142 			changeLeft = (options.to.left !== undefined) ? (options.to.left - startLeft) : 0;
143 			startTop = parseInt(options.el.style.top.replace(/px|em|pt/,""), 10) || 0;
144 			changeTop = (options.to.top !== undefined) ? (options.to.top - startTop) : 0;
145 
146 			return movesScroll.apply( this, [ startLeft, startTop, changeLeft, changeTop, options ] );
147 		};

149         /* documented in antie.devices.device */
150 		Device.prototype.hideElement = function(options) {
151 		    var animationDefaults;
152 			if (this.getConfig().animationDisabled || options.skipAnim) {
153 				options.el.style.visibility = "hidden";
154 				options.el.style.opacity = 0;
155 				if (typeof options.onComplete === "function") {
156 					options.onComplete();
157 				}
158                 return null;
159 			} else {
160 				animationDefaults = this.getConfig().defaults && this.getConfig().defaults.hideElementFade || {};
161 				return this._tween({
162 					el: options.el,
163 					style: options.el.style,
164 					from: {
165                         opacity: isNaN(parseInt(options.el.style.opacity, 10)) ? 1 : parseFloat(options.el.style.opacity)
166 					},
167 					to: {
168 						opacity: 0
169 					},
170 					fps: options.fps || animationDefaults.fps || 25,
171 					duration: options.duration || animationDefaults.duration || 840,
172 					easing: options.easing || animationDefaults.easing || 'linear',
173 					onComplete: function() {
174 						options.el.style.visibility = 'hidden';
175 						if (options.onComplete) {
176 							options.onComplete();
177 						}
178 					}
179 				});
180 			}
181 		};

184 		Device.prototype.showElement = function(options) {
185 		    var animationDefaults;
186 			if (this.getConfig().animationDisabled || options.skipAnim) {
187 				options.el.style.visibility = "visible";
188 				options.el.style.opacity = 1;
189 				if (typeof options.onComplete === "function") {
190 					options.onComplete();
191 				}
192                 return undefined;
193 			} else {
194 				animationDefaults = this.getConfig().defaults && this.getConfig().defaults.showElementFade || {};
195 				return this._tween({
196 					el: options.el,
197 					style: options.el.style,
198 					from: {
199                         opacity : isNaN(parseInt(options.el.style.opacity, 10)) ? 0 : parseFloat(options.el.style.opacity)
200 					},
201 					to: {
202 						opacity: 1
203 					},
204 					fps: options.fps || animationDefaults.fps || 25,
205 					duration: options.duration || animationDefaults.duration || 840,
206 					easing: options.easing || animationDefaults.easing || 'linear',
207 					onComplete: options.onComplete,
208 					onStart: function () {
209 						options.el.style.visibility = 'visible';
210 					}
211 				});
212 			}
213 		};

215         Device.prototype.tweenElementStyle = function(options) {
216             
217             var endPoints;
218 
219             function fireComplete() {
220                 if(options.onComplete) {
221                     options.onComplete();
222                 }
223             }
224 
225             function skipAnimation() {
226                 var i, properties, property;
227                 properties = endPoints.getProperties();
228                 for(i = 0; i !== properties.length; i += 1) {
229                     property = properties[i];
230                     options.el.style[property] = endPoints.getPropertyDestination(property);
231                 }
232             }
233             
234             function getTweenOptions(){
235                 var i, properties, property, tweenOptions;
236                 properties = endPoints.getProperties();
237                 tweenOptions = {
238                     el: options.el,
239                     style: options.el.style,
240                     fps: options.el.fps,
241                     duration: options.duration,
242                     easing: options.easing,
243                     onComplete: options.onComplete
244                 };
245                 tweenOptions.to = {};
246                 tweenOptions.from = {};
247                 
248                 for(i = 0; i !== properties.length; i += 1) {
249                     property = properties[i];
250                     tweenOptions.to[property] = endPoints.getPropertyDestination(property);
251                     tweenOptions.from[property] = endPoints.getPropertyOrigin(property);
252                 }
253                 return tweenOptions;
254             }
255             
256             endPoints = new TransitionEndPoints(options);
257             endPoints.completeOriginsUsingElement(options.el);
258             
259             if(endPoints.toAndFromAllEqual()) {
260                 fireComplete();
261                 return null;
262             } 
263             
264             if(options.skipAnim  || this.getConfig().animationDisabled) {
265                 skipAnimation();
266                 fireComplete();
267                 return undefined;
268             }
269             
270             return this._tween(getTweenOptions());
271         };
